# Crea wrapper
cat > ~/.argos-playwright-wrapper.sh << 'EOF'
#!/bin/bash
echo "[ARGOS Playwright] Avvio..." >&2
while true; do
  npx -y @playwright/mcp@latest "$@"
  echo "[ARGOS Playwright] Terminato. Riavvio in 2s..." >&2
  sleep 2
done
EOF
chmod +x ~/.argos-playwright-wrapper.sh
